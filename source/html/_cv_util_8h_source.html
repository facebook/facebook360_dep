226 template <typename T>
227 inline cv::Mat_<T> convertImage(const cv::Mat& imageIn) {
228  // Create a dummy mat that will hold type, depth and channels for a given T
229  cv::Mat_<T> infoMat;
230 
231  // Convert to desired depth
232  cv::Mat imageOut = convertTo(imageIn, infoMat.depth());
233 
234  // Special case: OpenCV doesn't have a bool type, so Mat_<bool> is CV_8U [0..255]
235  // We need to force values to be in [0..1]
236  if (std::is_same<T, bool>::value) {
237  cv::threshold(imageOut, imageOut, 127, 1, cv::THRESH_BINARY);
238  }
239 
240  // Convert to desired number of channels
241  const int chI = imageIn.channels();
242  const int chO = infoMat.channels();
243  if (chI == chO) {
244  return imageOut;
245  }
246 
247  if (chI == 1 && chO == 3) {
248  cv::cvtColor(imageOut, imageOut, cv::COLOR_GRAY2BGR);
249  } else if (chI == 1 && chO == 4) {
250  cv::cvtColor(imageOut, imageOut, cv::COLOR_GRAY2BGRA);
251  } else if (chI == 3 && chO == 1) {
252  cv::cvtColor(imageOut, imageOut, cv::COLOR_BGR2GRAY);
253  } else if (chI == 3 && chO == 4) {
254  cv::cvtColor(imageOut, imageOut, cv::COLOR_BGR2BGRA);
255  } else if (chI == 4 && chO == 1) {
256  cv::cvtColor(imageOut, imageOut, cv::COLOR_BGRA2GRAY);
257  } else if (chI == 4 && chO == 3) {
258  cv::cvtColor(imageOut, imageOut, cv::COLOR_BGRA2BGR);
259  } else {
260  CHECK(false) << "Conversion from " << chI << " channels to " << chO
261  << " channels not supported";
262  }
263 
264  return imageOut;
265 }

335 inline cv::Mat_<float> maskedMedianBlur(
336  const cv::Mat_<float>& mat,
337  const cv::Mat_<float>& background,
338  const cv::Mat_<bool>& mask,
339  const int radius,
340  const bool ignoreNan = true) {
341  cv::Mat_<float> blurred(mat.size(), 0.0);
342  for (int y = 0; y < mat.rows; ++y) {
343  for (int x = 0; x < mat.cols; ++x) {
344  std::vector<float> values;
345  if (!mask(y, x)) {
346  if (!background.empty()) {
347  blurred(y, x) = background(y, x);
348  }
349  continue;
350  }
351  for (int yy = y - radius; yy <= y + radius; ++yy) {
352  for (int xx = x - radius; xx <= x + radius; ++xx) {
353  // Ignore out of bounds
354  if (0 > yy || yy >= mat.rows || 0 > xx || xx >= mat.cols) {
355  continue;
356  }
357 
358  // Ignore outside mask
359  if (!mask(yy, xx)) {
360  continue;
361  }
362 
363  // Ignore NAN values if specified to do so
364  if (ignoreNan && (std::isnan(mat(yy, xx)) || mat(yy, xx) == 0)) {
365  continue;
366  }
367 
368  values.push_back(mat(yy, xx));
369  }
370  }
371 
372  if (values.size() != 0) {
373  const size_t n = values.size() / 2;
374  std::partial_sort(values.begin(), values.begin() + n + 1, values.end());
375  if (values.size() % 2 == 1) {
376  blurred(y, x) = values[n];
377  } else {
378  blurred(y, x) = (values[n - 1] + values[n]) / 2.0;
379  }
380  }
381  }
382  }
383  return blurred;
384 }

15  const Camera::Vector3& rotation,
16  const Camera::Vector3& translation,
17  const Eigen::UniformScaling<double>& scale,
18  const bool applyInReverse = false) {
19  const Eigen::AngleAxis<Camera::Real> x(rotation.x(), Camera::Vector3::UnitX());
20  const Eigen::AngleAxis<Camera::Real> y(rotation.y(), Camera::Vector3::UnitY());
21  const Eigen::AngleAxis<Camera::Real> z(rotation.z(), Camera::Vector3::UnitZ());
22  const Eigen::Affine3d r = Eigen::Affine3d(z * y * x);
23 
24  const Eigen::Translation3d t(translation);
25 
26  // Create transform from rotation, position and scale
27  Eigen::Transform<double, 3, Eigen::Affine> xform;
28  if (applyInReverse) {
29  xform = r * t * scale;
30  } else {
31  xform = scale * t * r;

33  return xform;

105 Camera::Rig transformRig(
106  const Camera::Rig& rig,
107  const Camera::Vector3& rotation,
108  const Camera::Vector3& translation,
109  const Eigen::UniformScaling<double>& scale,
110  const bool applyInReverse = false) {
111  Camera::Rig result;
112 
113  // Generate a transform with rotation only (no translation or scaling) for the forward, up and
114  // right vectors
115  const Eigen::Transform<double, 3, Eigen::Affine> rot = generateTransform(
116  rotation, Camera::Vector3(0, 0, 0), Eigen::UniformScaling<double>(1), applyInReverse);
117  const Eigen::Transform<double, 3, Eigen::Affine> xform =
118  generateTransform(rotation, translation, scale, applyInReverse);
119 
120  for (Camera camera : rig) {
121  const Camera::Vector3 forward = camera.forward();
122  const Camera::Vector3 up = camera.up();
123  const Camera::Vector3 right = camera.right();
124  camera.setRotation(rot * forward, rot * up, rot * right);
125  camera.position = xform * camera.position;
126  result.push_back(camera);
127  }
128  return result;
129 }

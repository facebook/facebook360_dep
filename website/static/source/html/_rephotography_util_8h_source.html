28 // Assuming x, y in [0, 1]
29 // SSIM = L^a * C^b + S^g, where a, b, g > 0 and
30 // L = (2 * muX * muY + c1) / (muX^2 + muY^2 + c1)
31 // C = (2 * sigmaXY + c2) / (sigmaX^2 + sigmaY^2 + c2)
32 // S = (sigmaXY + c3) / (sigmaX * sigmaY + c3)
33 //
34 // Note that NCC = SSIM wih a = 0, b = 0, g = 1
35 //
36 // NOTE: this will not work with non floating point images
37 template <typename T>
38 inline cv::Mat_<T> computeSSIM(
39  const cv::Mat_<T>& x,
40  const cv::Mat_<T>& y,
41  const int blurRadius,
42  const float alpha = 1.0f,
43  const float beta = 1.0f,
44  const float gamma = 1.0f) {
45  CHECK_EQ(x.size(), y.size());
46  CHECK_EQ(x.channels(), 3); // RGB
47  CHECK_EQ(x.channels(), y.channels());
48  CHECK_GT(blurRadius, 0);
49  CHECK_GE(alpha, 0.0f);
50  CHECK_GE(beta, 0.0f);
51  CHECK_GE(gamma, 0.0f);
52 
53  const cv::Mat_<T> muX = blur(x, blurRadius);
54  const cv::Mat_<T> muY = blur(y, blurRadius);
55  const cv::Mat_<T> mu2X = muX.mul(muX);
56  const cv::Mat_<T> mu2Y = muY.mul(muY);
57  const cv::Mat_<T> muXY = muX.mul(muY);
58  const cv::Mat_<T> xMuX2 = (x - muX).mul(x - muX);
59  const cv::Mat_<T> sig2X = blur(xMuX2, blurRadius);
60  const cv::Mat_<T> yMuY2 = (y - muY).mul(y - muY);
61  const cv::Mat_<T> sig2Y = blur(yMuY2, blurRadius);
62  const cv::Mat_<T> xMuxYMuy = (x - muX).mul(y - muY);
63  const cv::Mat_<T> sigXY = blur(xMuxYMuy, blurRadius);
64  cv::Mat_<T> sigX;
65  cv::sqrt(sig2X, sigX);
66  cv::Mat_<T> sigY;
67  cv::sqrt(sig2Y, sigY);
68 
69  // Default constants in the SSIM index formula: K = [0.01 0.03], L = 1
70  // c1 = (k * L)^2, k = 0.01, L = 1
71  // c2 = (k * L)^2, k = 0.03, L = 1
72  // c3 = c2 / 2
73  const cv::Scalar c1 = cv::Scalar::all(0.0001f);
74  const cv::Scalar c2 = cv::Scalar::all(0.0009f);
75  const cv::Scalar c3 = c2 / 2.0f;
76 
77  cv::Mat_<T> luminance = (2 * muXY + c1).mul(1.0f / (mu2X + mu2Y + c1));
78  cv::pow(luminance, alpha, luminance);
79  cv::Mat_<T> contrast = (2 * sigX.mul(sigY) + c2).mul(1.0f / (sig2X + sig2Y + c2));
80  cv::pow(contrast, beta, contrast);
81  cv::Mat_<T> structure = (sigXY + c3).mul(1.0f / (sigX.mul(sigY) + c3));
82  cv::pow(structure, gamma, structure);
83  return contrast.mul(luminance).mul(structure);
84 }

86 inline cv::Scalar averageScore(const cv::Mat& scoreMap, const cv::Mat& mask = cv::Mat()) {
87  cv::Scalar result;

89  cv::split(scoreMap, channels);

134

138  images.push_back(ref);
139  }

141  cv::Mat_<cv::Vec3b> ren = cv_util::convertImage<cv::Vec3b>(rendered[i]);
142  if (!mask.empty()) {
143  ren.setTo(cv::Scalar::all(0), ~mask);
144  }
145  images.push_back(ren);
146  }
147

150

153  ssim8 = cv::Scalar::all(255) - ssim8;
154  cv::applyColorMap(ssim8, ssim8, cv::COLORMAP_JET);

156  // Mask pixels outside mask
157  ssim8.setTo(cv::Scalar::all(0), ~mask);
158

161

164  const int textFont = cv::FONT_HERSHEY_PLAIN;
165  const double textScale = 2;

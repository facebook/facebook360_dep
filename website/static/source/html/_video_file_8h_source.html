49  void readBegin(const RigScene& scene, bool cull = false) {
50  const folly::dynamic& frame = catalog["frames"][frames[current]];
51  pending.emplace_back();
52  std::vector<Loader>& loaders = pending.back();
53  // kick off a loader for every camera in scene.rig
54  loaders.reserve(scene.rig.size());
55  for (int i = 0; i < int(scene.rig.size()); ++i) {
56  const Camera& camera = scene.rig[i];
57  const folly::dynamic& layout = frame[camera.id];
58  if (cull && i < int(scene.culled.size()) && scene.culled[i]) {
59  loaders.push_back({nullptr, 0, 0, layout, nullptr});
60  loaders.back().read = nullptr;
61  } else {
62  const uint64_t size = layout["size"].getInt();
63  // when reading, size must be page aligned
64  const uint64_t sizeAligned = align(size, kPageSize);
65  // allocate, map and align a buffer
66  const uint64_t sizeAlloc = sizeAligned + kPageSize - 1;
67  const GLuint buffer = createBuffer(kBufferType, (uint8_t*)nullptr, sizeAlloc);
68  uint8_t* const p = static_cast<uint8_t*>(glMapBuffer(kBufferType, GL_WRITE_ONLY));
69  CHECK(p);
70  uint8_t* const pAligned = align(p, kPageSize);
71  glBindBuffer(kBufferType, 0);
72  // start the read
73  const uint64_t offset = layout["offset"].getInt();
74  StripedFile::PendingRead* const read = stripedFile.readBegin(pAligned, offset, sizeAligned);
75  // stash the loader information for this camera
76  const uint64_t offsetUnaligned = offset - (pAligned - p);
77  loaders.push_back({read, buffer, offsetUnaligned, layout, p});
78  }
79  }
80  // increment frame counter
81  current = (current + 1) % frames.size();
82  }

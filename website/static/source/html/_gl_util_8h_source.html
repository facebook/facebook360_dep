55 inline Eigen::Projective3f
56 frustum(float minX, float maxX, float minY, float maxY, float minZ, float maxZ = INFINITY) {
57  // see glFrustum: http://goo.gl/fsgoMf
58  float A = (maxX + minX) / (maxX - minX);
59  float B = (maxY + minY) / (maxY - minY);
60  float C = -(maxZ + minZ) / (maxZ - minZ);
61  float D = -2 * maxZ * minZ / (maxZ - minZ);
62 
63  // fix special case maxZ == inf
64  if (std::isinf(maxZ)) {
65  C = -1;
66  D = -2 * minZ;
67  }
68 
69  Eigen::Matrix4f m;
70  m << 2 * minZ / (maxX - minX), 0, A, 0, 0, 2 * minZ / (maxY - minY), B, 0, 0, 0, C, D, 0, 0, -1,
71  0;
72 
73  return Eigen::Projective3f(m);
74 }

76 // produce a triangle strip covering a width x height grid
77 inline std::vector<GLuint> stripify(int width, int height, int skip = 1) {
78  std::vector<GLuint> result;
79  for (int y = 0; y + skip < height; y += skip) {
80  result.push_back(y * width); // double-hit the first index
81  for (int x = 0; x < width; x += skip) {
82  result.push_back(y * width + x);
83  result.push_back((y + skip) * width + x);
84  }
85  result.push_back(result.back()); // double-hit the last index
86  }
87  return result;
